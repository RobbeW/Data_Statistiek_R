\documentclass[dvipsnames, border=1cm,tikz]{standalone}
\usetikzlibrary {arrows.meta,bending,positioning}

\def\Only<#1-#2>#3{\ifnum\numexpr\X+1>#1
	\ifnum\numexpr\X-1<#2
	\pgfmathtruncatemacro{\DeltaX}{\X+1-#1}
	\xdef\DeltaX{\DeltaX}
	#3
	\fi\fi}

\begin{document}
	\pgfmathtruncatemacro{\Xmax}{110}
	
	\foreach \X in {0,1,...,\Xmax}
	{\begin{tikzpicture}[
				fat line/.style={
					line width=0.12cm,
					red,
					{Circle}-{Circle},
					shorten >=-2*\pgflinewidth,
					shorten <=-2*\pgflinewidth},
				cc/.style={
					line width=0.11cm,
					NavyBlue},annot/.style={scale=5}]
			\draw[line width=0.05cm,-{Stealth[length=5mm]}] (-6,0) -- (6,0) node[right] {\Huge $x$};
			\draw[line width=0.05cm,-{Stealth[length=5mm]}] (0,-6) -- (0,6) node[above] {\Huge $y$};
			%
			\Only<2-10>{
				\draw[fat line] (0,0) -- (\X/2,0);
			}
			\Only<11-20>{
				\draw[fat line] (0,0) -- (5,0) node[midway,above,annot,red]{\textsf{r}};
			}
			\Only<21-50>{
				\typeout{Delta=\DeltaX}
				\draw[cc] (5,0) arc[start angle=0,end angle={\DeltaX*12},radius=5];
				\draw[fat line] (0,0) -- ({\DeltaX*12}:5);
			}
			\Only<91-109>{
				\fill [LimeGreen,fill opacity=0.2] (0,0) -- (2.5,0) node[LimeGreen,below,annot,opacity=1, xshift=-0.04cm]{\textsf{1 rad}} arc[start angle=0, end angle={180/pi}, radius=2.5] -- cycle;
				\draw[cc,LimeGreen] (5,0) -- (0,0) -- ({180/pi}:5);
			}
			\Only<51-\Xmax>{
				\draw[cc] circle[radius=5cm];
			}
			\Only<51-70>{
				\draw[fat line] (5,0)  --++ ({180-\DeltaX*4.5}:5);
			}
			\Only<71-90>{
				\draw[fat line] (5,0) arc (0:{9*\DeltaX/pi}:5) --++ ({9*\DeltaX/pi+90}:{(20-\DeltaX)/4});
			}
		\end{tikzpicture}}
\end{document}
